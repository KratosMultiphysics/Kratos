set(CMAKE_INCLUDE_CURRENT_DIR ON)

message("**** configuring KratosBloodFlowApplication ****")

include_directories( ${CMAKE_SOURCE_DIR}/kratos )

## generate variables with the sources
set( KRATOS_BLOOD_FLOW_APPLICATION_SOURCES     
	${CMAKE_CURRENT_SOURCE_DIR}/blood_flow_application.cpp 
	${CMAKE_CURRENT_SOURCE_DIR}/custom_elements/artery_element.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/custom_conditions/artery_condition.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/custom_conditions/artery_inlet_condition.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/custom_conditions/artery_outlet_condition.cpp
        ${CMAKE_CURRENT_SOURCE_DIR}/custom_python/blood_flow_python_application.cpp
	${CMAKE_CURRENT_SOURCE_DIR}/custom_python/add_custom_utilities_to_python.cpp
)

###############################################################
## define library Kratos which defines the basic python interface
add_library(KratosBloodFlowApplication SHARED ${KRATOS_BLOOD_FLOW_APPLICATION_SOURCES})
target_link_libraries(KratosBloodFlowApplication KratosCore )
set_target_properties(KratosBloodFlowApplication PROPERTIES PREFIX "")
install(TARGETS KratosBloodFlowApplication DESTINATION libs )

# changing the .dll suffix to .pyd 
if(${CMAKE_SYSTEM_NAME} MATCHES "Windows")
	set_target_properties(KratosBloodFlowApplication PROPERTIES SUFFIX .pyd)
endif(${CMAKE_SYSTEM_NAME} MATCHES "Windows")

if(${INSTALL_PYTHON_FILES} MATCHES ON)
  get_filename_component (CURRENT_DIR_NAME ${CMAKE_CURRENT_SOURCE_DIR} NAME)
  install(DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}/python_scripts DESTINATION applications/${CURRENT_DIR_NAME}  FILES_MATCHING PATTERN "*.py"  PATTERN ".svn" EXCLUDE)
endif(${INSTALL_PYTHON_FILES} MATCHES ON)

# Add to the KratosMultiphisics Python module
install(FILES "${CMAKE_CURRENT_SOURCE_DIR}/BloodFlowApplication.py" DESTINATION KratosMultiphysics )
